ARCH LINUX INSTALLATION:
-Set the keyboard layout
	loadkeys la-latin1
-Verifying Internet Connection (Ethernet)
	ip link
	ping google.com (verifying connection. Ctrl+C for stop)
-Update the system clock:
	timedatectl set-ntp true
	///timedatectl set-timezone America/Bogota (this not)
	timedatectl status (Verifying update)
-Partition Disk
	lsblk
	cfdisk (Delete all partitions)
	fdisk /dev/sda
	o
	n
	*enter*
	*enter*
	+80G
	n
	*enter*
	*enter*
	+216.1G
	n
	*enter*
	*enter*
	*enter*
	w
	lsblk
	cfdisk and do bootable the first parition
	mkfs.ext4 /dev/sda1
	mkfs.ext4 /dev/sda2
	mount /dev/sda1 /mnt
	mkdir /mnt/home
	mount /dev/sda2 /mnt/home
	mkswap /dev/sda3
	swapon /dev/sda3
-Select the mirrors
	nano /etc/pacman.d/mirrorlist (Select close mirrors servers)
-INSTALL
	pacstrap /mnt base linux linux-firmware nano
CONFIGURE THE SYSTEM
-Fstab
	genfstab -U /mnt >> /mnt/etc/fstab
-Chroot
	arch-chroot /mnt
-Time zone
	ln -sf /usr/share/zoneinfo/America/Bogota /etc/localtime
	hwclock --systohc
-Localization
	locale-gen
	pacman -S nano (aveces hay que instalarlo)
	nano /etc/locale.gen (delete hashtag that be localizated before the language)
	locale-gen
	nano /etc/locale.conf
	LANG=en_US.UTF-8 (Ctrl+X and Yes)
	nano /etc/vconsole.conf
	KEYMAP=la-latin1 (Ctrl+x and Yes)
